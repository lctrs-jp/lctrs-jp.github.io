/**************************************/ // 補足課題3 // 2006/04/28 08033???? 西田直樹 // ex103.c // 入力：整数 // 機能：与えられた整数番目の // フィボナッチ数を表示 /**************************************/ #include int fib( int n ); int fastfib( int n ); /*-----------------------// メイン関数 (fibをテスト) /*-----------------------*/ main() { int num; /* 自然数を１つ獲得 */ printf("自然数を入力してください: "); scanf("%d", &num ); /* fib の結果を表示 */ if( num >= 0 ){ /* n が負でないとき */ printf("%d 番目のフィボナッチ数: %d\n", num, fastfib(num) ); }else{ printf("正の自然数を入力してください．( Fib(%d) = %d )\n", num, fastfib(num) ); } } /*------------------------------------*/ // 関 数：fastfib // 引 数：n (int) // 返り値：A_n (int) // 機 能：n 番目のフィボナッチ数A_nを返す /*------------------------------------*/ int fastfib( int n ) { int i; int A_n, A_1, A_0; if( n <= 0 ){ /* nが0以下のときは0を返す */ return( 0 ); }else if( n == 1 ){ /* nが1のときは1を返す */ return( 1 ); } A_0 = 0; /* Fib(n-2) の初期値を設定 */ A_1 = 1; /* Fib(n-1) の初期値を設定 */ for( i = 2 ; i <= n ; i++ ){ A_n = A_1 + A_0; /* Fib(i) を計算 */ /* i+1のためのFib(i-1)とFib(i)をセット */ A_0 = A_1; A_1 = A_n; } return( A_n ); } /*------------------------------------*/ // 関 数：fib // 引 数：n (int) // 返り値：res (int) // 機 能：n 番目のフィボナッチ数resを返す /*------------------------------------*/ int fib( int n ) { int res; if( n <= 0 ){ /* nが0以下のときは返り値を0に */ res = 0; }else if( n == 1 ){ /* nが1のときは返り値を1に */ res = 1; }else{ /* そうでないとき */ res = fib( n-1 ) + fib( n-2 ); /* 再帰呼出でn番目の値を計算 */ } return( res ); }
