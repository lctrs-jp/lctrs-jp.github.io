Exercise 4 (October 23, 2014)

Consider the following program: 

#include ⟨stdio.h⟩

int main(void)
{
  int temp[31], i, min, max, avg;
  int days;

  printf("How many days in the month? ");
  scanf("%d", &days);

  for (i = 0; i < days; i++) {
    printf("Input the temperature of the %d-th day at noon: ", i + 1);
    scanf("%d", &temp[i]);
  }

  /* Compute the average */
  avg = 0;
  for (i = 0; i < days; i++)
    avg = avg + temp[i];
  printf("Ave.: %d C\n", avg / days);

  /* Compute the maximum and minimum temperatures */
  min = 200;  /* initialize min and max */
  max = 0;
  for (i = 0; i < days; i++) {
    if (min > temp[i])
      min = temp[i];
    if (max < temp[i])
      max = temp[i];
  }

  printf("Max.: %d C\n", min);
  printf("Min.: %d C\n", max);

  return 0;
}
Revise this program, adapting to the following requirement:
  • the program receives the maximum temperature of the day as real with type double,
  • the progrram deals with the data of a week (i.e., fixes the number of days to seven), and
  • in addition to the max., min., and ave. temperatures, the program shows the standard deviation, the variance, and the median.
Then, execute the program with the maximum temperatures in Nagoya from Oct. 1 - 7, 2014 (see http://bit.ly/1a7KZQ3).

Submit the source and the result of the execution to NUCT. Note that you don't have to use your personal data in executing.

Example of Solutions

#include ⟨stdio.h⟩
#include ⟨math.h⟩

int main(void)
{
  int i, j, days = 7;
  double temp[7], min, max, avg, ss, t;

  for (i = 0; i < days; i++) {
    printf("Input the temperature of the %d-th day at noon: ", i + 1);
    scanf("%lf", &temp[i]);
  }

  /* Compute the average */
  avg = 0;
  min = 200;  /* initialize min and max */
  max = 0;
  for (i = 0; i < days; i++){
    avg = avg + temp[i];
    if (min > temp[i])
      min = temp[i];
    if (max < temp[i])
      max = temp[i];
  }
  avg = avg / days; /* store the average */
  printf("  Ave.: %lf\n", avg);
  printf("  Min.: %lf\n", min);
  printf("  Max.: %lf\n", max);

  /* Compute the variance */
  ss = 0;
  for(i = 0; i < days; i++)
    ss = ss + pow(temp[i] - avg, 2);
  ss = ss / days;
  printf("Std. Deviation: %lf\n", sqrt(ss) );
  printf("      Variance: %lf\n", ss );

  /* to compute the median, sort the array */
  for (i = 1; i < days; ++i)
    for (j = days - 1; j >= i; --j) {
      if (temp[j - 1] > temp[j]) {
        t = temp[j - 1];
        temp[j - 1] = temp[j];
        temp[j] = t;
      }
    }
  /* show the 4th element which is the median */
  printf("  Median: %lf\n", temp[3] );

  return 0;
}

Exercise 5 (October 23, 2014)

Implement strcpy, strcat, and strlen yourself as mystrcpy, mystrcat, and mystrlen, resp., with the following specification:

  • void mystrcpy(char trg[], char src[]);
  • void mystrcat(char trg[], char src[]);
  • int mystrlen(char trg[]);
Then, test them by using the following program: 
#include ⟨string.h⟩
#include ⟨stdio.h⟩

void mystrcpy(char trg[], char src[]);
void mystrcat(char trg[], char src[]);
int mystrlen(char trg[]);
int mystrcmp(char trg[], char src[]);

int main(void)
{
  char str1[80], str2[80];
  int i;

  printf("Input the first string: ");
  gets(str1);
  printf("Input the second string: ");
  gets(str2);

  /* check the length of the strings */
  printf("The length of %s is %d.\n", str1, mystrlen(str1));
  printf("The length of %s is %d.\n", str2, mystrlen(str2));

  /* compare the strings */
  i = mystrcmp(str1, str2);
  if (!i)
    printf("These strings are equivalent.\n");
  else if (i < 0)
    printf("%s is less than %s.\n", str1, str2);
  else
    printf("%s is greater than %s.\n", str1, str2);

  /* concatenate the strings if the empty space is enough */
  if ((mystrlen(str1) + mystrlen(str2)) < 80) {
    mystrcat(str1, str2);
    printf("%s\n", str1);
  }

  /* copy str2 to str1 */
  mystrcpy(str1, str2);
  printf("%s %s\n", str1, str2);

  return 0;
}

void mystrcpy(char trg[], char src[])
{
  int i;

  for( i = 0 ; src[i] != '\0' ; i++ )
    trg[i] = src[i];
  trg[i] = '\0';

  return;
}

void mystrcat(char trg[], char src[])
{
  int i, j;

  for( i = 0; trg[i] != '\0' ; i++ );
      /* or i = mystrlen(trg); */

  for( j = 0 ; src[j] != '\0' ; j++ ){
    trg[i] = src[j];
    i++;
  }
  trg[i] = '\0';

  return;
}

int mystrlen(char trg[])
{
  int len = 0;
  while( trg[len] != '\0' ){
    len ++;
  }
  return len;
}

int mystrcmp(char trg[], char src[])
{
  int i;

  for( i = 0 ; trg[i] == src[i] ; i++ ){
    if( trg[i] == '\0' ) break;
  }

  return trg[i] - src[i];
}
