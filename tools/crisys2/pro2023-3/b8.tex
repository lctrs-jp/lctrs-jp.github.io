\paragraph{LCTRS}

\[

\Sigmainttheory \cup\left\{
\begin{array}{r@{\>}c@{\>}l}

\symb{noncrit1} & : & \Sort{Loc}\\
\symb{wait1} & : & \Sort{Loc}\\
\symb{crit1} & : & \Sort{Loc}\\
\symb{noncrit2} & : & \Sort{Loc}\\
\symb{wait2} & : & \Sort{Loc}\\
\symb{crit2} & : & \Sort{Loc}\\

\end{array}

\[
\left\{
\begin{array}{rr@{\>}c@{\>}ll}

 & \to

\mathrm{3} :

 & \Constraint{\var{y11} = \symb{0}}\\

\mathrm{4} : 
 & \symb{state}(\var{p1},\symb{noncrit2},\var{y1},\var{y2})
 & \to 
 & \symb{state}(\var{p1},\symb{wait2},\var{y1},\var{y21})
 & \Constraint{\var{y21} = (\var{y1} + \symb{1})}\\

\mathrm{5} : 
 & \symb{state}(\var{p1},\symb{wait2},\var{y1},\var{y2})
 & \to 
 & \symb{state}(\var{p1},\symb{crit2},\var{y1},\var{y2})
 & \Constraint{\var{y1} = \symb{0} \lor (\var{y2} < \var{y1})}\\

\mathrm{6} : 
 & \symb{state}(\var{p1},\symb{crit2},\var{y1},\var{y2})
 & \to

\mathrm{7} :

\mathrm{8} :
