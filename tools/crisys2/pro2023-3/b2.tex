\section*{Original LCTRSs}

\paragraph{LCTRS}

\[
\Sigma = 
\Sigmainttheory \cup\left\{
\begin{array}{r@{\>}c@{\>}l}
\symb{state} & : & \Sort{Loc} \times \Sort{Loc} \times \Sort{int} \Rightarrow \Sort{State}\\
\symb{noncrit1} & : & \Sort{Loc}\\
\symb{wait1} & : & \Sort{Loc}\\
\symb{crit1} & : & \Sort{Loc}\\
\symb{noncrit2} & : & \Sort{Loc}\\
\symb{wait2} & : & \Sort{Loc}\\
\symb{crit2} & : & \Sort{Loc}\\
\symb{success} & : & \Sort{State}\\
\symb{error} & : & \Sort{State}\\
\end{array}
\right\}
\]

\[
\left\{
\begin{array}{rr@{\>}c@{\>}ll}
\mathrm{1} : 
 & \symb{state}(\symb{noncrit1},\var{p2},\var{y})
 & \to 
 & \symb{state}(\symb{wait1},\var{p2},\var{y})
 & \Constraint{\var{y} > \symb{0}}\\

\mathrm{2} : 
 & \symb{state}(\symb{wait1},\var{p2},\var{y})
 & \to 
 & \symb{state}(\symb{crit1},\var{p2},\var{y1})
 & \Constraint{\var{y1} = (\var{y} - \symb{1})}\\

\mathrm{3} : 
 & \symb{state}(\symb{crit1},\var{p2},\var{y})
 & \to 
 & \symb{state}(\symb{noncrit1},\var{p2},\var{y1})
 & \Constraint{\var{y1} = (\var{y} + \symb{1})}\\

\mathrm{4} : 
 & \symb{state}(\var{p1},\symb{noncrit2},\var{y})
 & \to 
 & \symb{state}(\var{p1},\symb{wait2},\var{y})
 & \Constraint{\var{y} > \symb{0}}\\

\mathrm{5} : 
 & \symb{state}(\var{p1},\symb{wait2},\var{y})
 & \to 
 & \symb{state}(\var{p1},\symb{crit2},\var{y1})
 & \Constraint{\var{y1} = (\var{y} - \symb{1})}\\

\mathrm{6} : 
 & \symb{state}(\var{p1},\symb{crit2},\var{y})
 & \to 
 & \symb{state}(\var{p1},\symb{noncrit2},\var{y1})
 & \Constraint{\var{y1} = (\var{y} + \symb{1})}\\

\mathrm{7} : 
 & \symb{state}(\var{p1},\var{p2},\var{y})
 & \to 
 & \symb{success}\\

\mathrm{8} : 
 & \symb{state}(\symb{crit1},\symb{crit2},\var{y})
 & \to 
 & \symb{error}\\

\end{array}
\right\}
\]
