\section*{Original LCTRSs}

\paragraph{LCTRS}

\[
\Sigma = 
\Sigmainttheory \cup\left\{
\begin{array}{r@{\>}c@{\>}l}
\symb{state} & : & \Sort{Loc} \times \Sort{Loc} \times \Sort{bool} \times \Sort{bool} \times \Sort{int} \Rightarrow \Sort{State}\\
\symb{noncrit0} & : & \Sort{Loc}\\
\symb{wait0} & : & \Sort{Loc}\\
\symb{crit0} & : & \Sort{Loc}\\
\symb{noncrit1} & : & \Sort{Loc}\\
\symb{wait1} & : & \Sort{Loc}\\
\symb{crit1} & : & \Sort{Loc}\\
\symb{success} & : & \Sort{State}\\
\symb{error} & : & \Sort{State}\\
\end{array}
\right\}
\]

\[
\left\{
\begin{array}{rr@{\>}c@{\>}ll}
\mathrm{1} : 
 & \symb{state}(\symb{noncrit0},\var{p1},\var{b0},\var{b1},\var{x})
 & \to 
 & \symb{state}(\symb{wait0},\var{p1},\var{b01},\var{b1},\var{x1})
 & \Constraint{\var{b01} \land \var{x1} = \symb{1}}\\

\mathrm{2} : 
 & \symb{state}(\symb{wait0},\var{p1},\var{b0},\var{b1},\var{x})
 & \to 
 & \symb{state}(\symb{crit0},\var{p1},\var{b0},\var{b1},\var{x})
 & \Constraint{\var{x} = \symb{0} \lor \neg \var{b1}}\\

\mathrm{3} : 
 & \symb{state}(\symb{crit0},\var{p1},\var{b0},\var{b1},\var{x})
 & \to 
 & \symb{state}(\symb{noncrit0},\var{p1},\var{b01},\var{b1},\var{x})
 & \Constraint{\neg(\var{b01})}\\

\mathrm{4} : 
 & \symb{state}(\var{p0},\symb{noncrit1},\var{b0},\var{b1},\var{x})
 & \to 
 & \symb{state}(\var{p0},\symb{wait1},\var{b0},\var{b11},\var{x1})
 & \Constraint{\var{b11} \land \var{x1} = \symb{0}}\\

\mathrm{5} : 
 & \symb{state}(\var{p0},\symb{wait1},\var{b0},\var{b1},\var{x})
 & \to 
 & \symb{state}(\var{p0},\symb{crit1},\var{b0},\var{b1},\var{x})
 & \Constraint{\var{x} = \symb{1} \lor \neg \var{b0}}\\

\mathrm{6} : 
 & \symb{state}(\var{p0},\symb{crit1},\var{b0},\var{b1},\var{x})
 & \to 
 & \symb{state}(\var{p0},\symb{noncrit1},\var{b0},\var{b11},\var{x})
 & \Constraint{\neg(\var{b11})}\\

\mathrm{7} : 
 & \symb{state}(\var{p0},\var{p1},\var{b0},\var{b1},\var{x})
 & \to 
 & \symb{success}\\

\mathrm{8} : 
 & \symb{state}(\symb{crit0},\symb{crit1},\var{b0},\var{b1},\var{x})
 & \to 
 & \symb{error}\\

\end{array}
\right\}
\]
