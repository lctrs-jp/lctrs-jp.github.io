\paragraph{LCTRS}

\[
\Sigma = 
\Sigmainttheory \cup\left\{
\begin{array}{r@{\>}c@{\>}l}
\symb{state} & : & \Sort{Loc} \times \Sort{Loc} \times \Sort{bool} \times \Sort{bool} \times \Sort{int} \Rightarrow \Sort{State}\\
\symb{c00} & : & \Sort{Loc}\\
\symb{c10} & : & \Sort{Loc}\\
\symb{c20} & : & \Sort{Loc}\\
\symb{crit0} & : & \Sort{Loc}\\
\symb{c01} & : & \Sort{Loc}\\
\symb{c11} & : & \Sort{Loc}\\
\symb{c21} & : & \Sort{Loc}\\
\symb{crit1} & : & \Sort{Loc}\\
\symb{success} & : & \Sort{State}\\
\symb{error} & : & \Sort{State}\\
\end{array}
\right\}
\]

\[
\left\{
\begin{array}{rr@{\>}c@{\>}ll}
\mathrm{1} : 
 & \symb{state}(\symb{c00},\var{p1},\var{b0},\var{b1},\var{k})
 & \to 
 & \symb{state}(\symb{c10},\var{p1},\var{b01},\var{b1},\var{k})
 & \Constraint{\neg(\var{b01})}\\

\mathrm{2} : 
 & \symb{state}(\symb{c10},\var{p1},\var{b0},\var{b1},\var{k})
 & \to 
 & \symb{state}(\symb{c20},\var{p1},\var{b0},\var{b1},\var{k})
 & \Constraint{\var{k} \ne \symb{0}}\\

\mathrm{3} : 
 & \symb{state}(\symb{c10},\var{p1},\var{b0},\var{b1},\var{k})
 & \to 
 & \symb{state}(\symb{crit0},\var{p1},\var{b0},\var{b1},\var{k})
 & \Constraint{\var{k} = \symb{0}}\\

\mathrm{4} : 
 & \symb{state}(\symb{c20},\var{p1},\var{b0},\var{b1},\var{k})
 & \to 
 & \symb{state}(\symb{c10},\var{p1},\var{b0},\var{b1},\var{k1})
 & \Constraint{\var{b1} \land \var{k1} = \symb{0}}\\

\mathrm{5} : 
 & \symb{state}(\symb{crit0},\var{p1},\var{b0},\var{b1},\var{k})
 & \to 
 & \symb{state}(\symb{c00},\var{p1},\var{b01},\var{b1},\var{k})
 & \Constraint{\var{b01}}\\

\mathrm{6} : 
 & \symb{state}(\var{p0},\symb{c01},\var{b0},\var{b1},\var{k})
 & \to 
 & \symb{state}(\var{p0},\symb{c11},\var{b0},\var{b11},\var{k})
 & \Constraint{\neg(\var{b11})}\\

\mathrm{7} : 
 & \symb{state}(\var{p0},\symb{c11},\var{b0},\var{b1},\var{k})
 & \to 
 & \symb{state}(\var{p0},\symb{c21},\var{b0},\var{b1},\var{k})
 & \Constraint{\var{k} \ne \symb{1}}\\

\mathrm{8} : 
 & \symb{state}(\var{p0},\symb{c11},\var{b0},\var{b1},\var{k})
 & \to 
 & \symb{state}(\var{p0},\symb{crit1},\var{b0},\var{b1},\var{k})
 & \Constraint{\var{k} = \symb{1}}\\

\mathrm{9} : 
 & \symb{state}(\var{p0},\symb{c21},\var{b0},\var{b1},\var{k})
 & \to 
 & \symb{state}(\var{p0},\symb{c11},\var{b0},\var{b1},\var{k1})
 & \Constraint{\var{b0} \land \var{k1} = \symb{1}}\\

\mathrm{10} : 
 & \symb{state}(\var{p0},\symb{crit1},\var{b0},\var{b1},\var{k})
 & \to 
 & \symb{state}(\var{p0},\symb{c01},\var{b0},\var{b11},\var{k})
 & \Constraint{\var{b11}}\\

\mathrm{11} : 
 & \symb{state}(\var{p0},\var{p1},\var{b0},\var{b1},\var{k})
 & \to 
 & \symb{success}\\

\mathrm{12} : 
 & \symb{state}(\symb{crit0},\symb{crit1},\var{b0},\var{b1},\var{k})
 & \to 
 & \symb{error}\\

\end{array}
\right\}
